[−][src]Macro some_macros::hash_set

macro_rules! hash_set {
    ( $(use HashSet<$t:ty>;)? $($e:expr),*$(,)? ) => { ... };
}

Constructs a HashSet, calling into on the values

Example:

let hm = hashset!(
    0,
    1,
    2
);
let mut control = HashSet::new();
control.insert(0);
control.insert(1);
control.insert(2);
assert_eq!(hm, control);

Optionally, you can start with the wanted type parameters

let hm = hash_set!(use HashSet<i32>;
    0u8,
    1,
    2
);
let mut control = HashSet::new();
control.insert(0);
control.insert(1);
control.insert(2);
assert_eq!(hm, control);
